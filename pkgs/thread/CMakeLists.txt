cmake_minimum_required(VERSION 3.20)
project(tclthread C)

# =====================================================================
# Pinned Tcl Thread version.  To upgrade, bump this one value.
# Releases: https://github.com/tcltk/thread/tags  (tag thread-X-Y-Z)
# =====================================================================
set(THREAD_VERSION 3.0.5)
# =====================================================================

# Tag form: 3.0.5 -> thread-3-0-5 ; windows nodots: 3.0.5 -> 305
string(REPLACE "." "-" _tv "${THREAD_VERSION}")
set(THREAD_TAG "thread-${_tv}")
string(REPLACE "." "" THREAD_NODOTS "${THREAD_VERSION}")

# ---------------------------------------------------------------------
# Fetch the thread extension source (TEA package; no CMakeLists, so
# MakeAvailable just populates the tree).
# ---------------------------------------------------------------------
include(FetchContent)
FetchContent_Declare(thread_src
    GIT_REPOSITORY https://github.com/tcltk/thread.git
    GIT_TAG        ${THREAD_TAG}
    GIT_SHALLOW    TRUE
    SOURCE_SUBDIR  do-not-configure
)
FetchContent_MakeAvailable(thread_src)

if(NOT EXISTS ${thread_src_SOURCE_DIR}/generic/threadCmd.c)
    message(FATAL_ERROR "thread source not populated: ${thread_src_SOURCE_DIR}")
endif()

# threadCmd.c #includes threadUuid.h (normally generated by the TEA
# build from manifest.uuid). Generate a minimal one.
file(WRITE ${CMAKE_CURRENT_BINARY_DIR}/threadUuid.h
    "#define THREAD_VERSION_UUID unknown\n")

# ---------------------------------------------------------------------
# Build the thread extension. The gdbm/lmdb persistent-store backends
# (psGdbm.c/psLmdb.c) are fully #ifdef-guarded and compile to nothing
# unless HAVE_GDBM/HAVE_LMDB are defined, so they add no external deps.
# ---------------------------------------------------------------------
set(THREAD_SRCS
    ${thread_src_SOURCE_DIR}/generic/threadNs.c
    ${thread_src_SOURCE_DIR}/generic/threadCmd.c
    ${thread_src_SOURCE_DIR}/generic/threadSvCmd.c
    ${thread_src_SOURCE_DIR}/generic/threadSpCmd.c
    ${thread_src_SOURCE_DIR}/generic/threadPoolCmd.c
    ${thread_src_SOURCE_DIR}/generic/psGdbm.c
    ${thread_src_SOURCE_DIR}/generic/psLmdb.c
    ${thread_src_SOURCE_DIR}/generic/threadSvListCmd.c
    ${thread_src_SOURCE_DIR}/generic/threadSvKeylistCmd.c
    ${thread_src_SOURCE_DIR}/generic/tclXkeylist.c
)
if(NOT WIN32)
    list(APPEND THREAD_SRCS ${thread_src_SOURCE_DIR}/unix/threadUnix.c)
endif()

add_library(tclthread SHARED ${THREAD_SRCS})

target_compile_definitions(tclthread PRIVATE
    USE_TCL_STUBS
    PACKAGE_VERSION="${THREAD_VERSION}"
)
if(WIN32)
    target_compile_definitions(tclthread PRIVATE BUILD_Thread)
endif()

target_include_directories(tclthread PRIVATE
    ${thread_src_SOURCE_DIR}/generic
    ${CMAKE_CURRENT_BINARY_DIR}          # threadUuid.h
)

# Tcl stub headers from the in-tree build
if(tcl_SOURCE_DIR)
    target_include_directories(tclthread PRIVATE
        ${tcl_SOURCE_DIR}/generic
        ${tcl_SOURCE_DIR}/libtommath
    )
    if(WIN32)
        target_include_directories(tclthread PRIVATE
            ${tcl_SOURCE_DIR}/win
            ${thread_src_SOURCE_DIR}/win
        )
    else()
        target_include_directories(tclthread PRIVATE ${tcl_SOURCE_DIR}/unix)
    endif()
endif()

target_link_libraries(tclthread PRIVATE tclstub)

set_target_properties(tclthread PROPERTIES POSITION_INDEPENDENT_CODE ON)

# Filenames must match the generated pkgIndex.tcl:
#   Windows : tcl9thread<nodots>.dll          (e.g. tcl9thread305.dll)
#   Unix    : libtcl9thread<version>.{so,dylib}  (e.g. libtcl9thread3.0.5.dylib)
if(WIN32)
    set_target_properties(tclthread PROPERTIES
        OUTPUT_NAME "tcl9thread${THREAD_NODOTS}"
        PREFIX ""
    )
else()
    set_target_properties(tclthread PROPERTIES
        OUTPUT_NAME "tcl9thread${THREAD_VERSION}"
        PREFIX "lib"
    )
    if(APPLE)
        set_target_properties(tclthread PROPERTIES LINK_FLAGS "-dynamiclib")
    endif()
endif()

# ---------------------------------------------------------------------
# Generate the version-stamped pkgIndex.tcl, and expose it plus the
# bundled ttrace.tcl to the parent scope for install_tcl_package().
# ---------------------------------------------------------------------
configure_file(
    ${CMAKE_CURRENT_SOURCE_DIR}/pkgIndex.tcl.in
    ${CMAKE_CURRENT_BINARY_DIR}/pkgIndex.tcl
    @ONLY
)
set(THREAD_TCL_FILES
    ${CMAKE_CURRENT_BINARY_DIR}/pkgIndex.tcl
    ${thread_src_SOURCE_DIR}/lib/ttrace.tcl
    PARENT_SCOPE)
